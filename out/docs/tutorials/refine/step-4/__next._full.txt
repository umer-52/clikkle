          authProvider={authProvider}

          notificationProvider={useNotificationProvider}

          <Routes>
            <Route
              element={
                <Authenticated

                  <ThemedLayoutV2>
                    <Outlet />
                  </ThemedLayoutV2>
                </Authenticated>
              }
            ></Route>

            <Route
              element={
                <Authenticated fallback={<Outlet />}>
                  <NavigateToResource

              <Route

                  <ThemedLayoutV2>

              }
            >
              <Route path="*" element={<ErrorComponent />} />
            </Route>
          </Routes>
        </Refine>
      </AntdApp>
    </ConfigProvider>
  </BrowserRouter>
